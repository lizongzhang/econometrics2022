1 一元回归

1.1 散点图上添加拟合线

#geom_smooth()添加回归线，method = lm 线性回归
mpg %>% 
  ggplot(aes(displ, cty))+
  geom_point()+
  geom_smooth(method = lm,
              se = F)
## `geom_smooth()` using formula 'y ~ x'

1.2 散点图上添加二阶曲线

#添加二阶曲线
mpg %>% 
  ggplot(aes(displ, cty))+
  geom_point()+
  geom_smooth(method = lm,
              se = F,
              formula = y ~ poly(x,2))+
  theme()

1.3 散点图添加主题，设置字体和颜色

#添加主题，设置字体和颜色

1.6 散点图标注回归方程

#plot()绘制散点图， 设置点的形状point character, pch=; 设置点的颜色 col = 
plot(mpg$displ, mpg$hwy, pch = 16, col = 5)

#Add Straight Lines to a Plot
abline(lm(mpg$hwy ~ mpg$displ), col = 9, lwd = 2)

#将lm(mpg$hwy ~ mpg$displ)的系数保留3位小数，保存到对象coef中
coef <- round(coef(lm(mpg$hwy ~ mpg$displ)), 3)

#text()在图中添加文本
#4，40代表添加的文本在图形中的横纵坐标
#paste()生成回归方程的表达式
#cex = 1.2 , character expansion, 代表相对于默认字体放大的倍数
text(4, 40,  paste("hwy-hat = ", coef[1], "+", coef[2], "displ"), cex = 1.2)

#方法二：ggplot中添加 
mpg %>% 
  ggplot(aes(displ, hwy))+
  geom_point()+
  geom_smooth(method = lm, se = F) +
  annotate("text", x = 4, y =40, 
           label = paste("hwy-hat = ", coef[1], "+", coef[2], "displ"), size = 6)+
  theme_bw()
## `geom_smooth()` using formula 'y ~ x'
