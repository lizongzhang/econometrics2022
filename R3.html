2 在散点图中添加回归方程及判定系数

#安装包ggpubr
# install.packages("ggpubr")

# 加载包ggpubr
library(ggpubr)

Table_2_14 %>% ggplot(aes(Age, Price))+
  geom_point()+
  geom_smooth(method = lm) +
  stat_regline_equation(
    aes(label = paste(..eq.label.., ..adj.rr.label.., sep = "~~~~")),
    formula = Table_2_14$Price ~ Table_2_14$Age)
